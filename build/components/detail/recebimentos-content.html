    $(document).ready(function() {

        $('.recebimentos-content__bg2').hide();

            $('.recebimentos-content__bg2').show();
            $('.recebimentos-content__form').hide();
        });

        $('.currency').mask("#.##0,00", {
            reverse: true
        });
    });
</script>
